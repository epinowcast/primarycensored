396vector primary_censored_sone_lpmf_vectorized(
397 int max_delay, data real D, int dist_id,
398 array[] real params, data real pwindow,
399 int primary_dist_id, array[] real primary_params
400) {
401
402 int upper_interval = max_delay + 1;
403 vector[upper_interval] log_pmfs;
404 vector[upper_interval] log_cdfs;
405 real log_normalizer;
406
407 // Check if D is at least max_delay + 1
408 if (D < upper_interval) {
409 reject("D must be at least max_delay + 1");
410 }
411
412 // Compute log CDFs
413 for (d in 1:upper_interval) {
414 log_cdfs[d] = primary_censored_dist_lcdf(
415 d | dist_id, params, pwindow, positive_infinity(), primary_dist_id,
416 primary_params
417 );
418 }
419
420 // Compute log normalizer using upper_interval
421 if (D > upper_interval) {
422 if (is_inf(D)) {
423 log_normalizer = 0; // No normalization needed for infinite D
424 } else {
425 log_normalizer = primary_censored_dist_lcdf(
426 D | dist_id, params, pwindow, positive_infinity(),
427 primary_dist_id, primary_params
428 );
429 }
430 } else {
431 log_normalizer = log_cdfs[upper_interval];
432 }
433
434 // Compute log PMFs
435 log_pmfs[1] = log_cdfs[1] - log_normalizer;
436 for (d in 2:upper_interval) {
437 log_pmfs[d] = log_diff_exp(log_cdfs[d], log_cdfs[d-1]) - log_normalizer;
438 }
439
440 return log_pmfs;
441}
