105real primary_censored_integrand(real x, real xc, array[] real theta,
106 array[] real x_r, array[] int x_i) {
107 // Unpack parameters
108 real d = theta[1];
109 int dist_id = x_i[1];
110 int primary_dist_id = x_i[2];
111 real pwindow = x_r[1];
112 real D = x_r[2];
113 int dist_params_len = x_i[3];
114 int primary_params_len = x_i[4];
115
116 // Extract distribution parameters
117 array[dist_params_len] real params;
118 if (dist_params_len) {
119 params = theta[2:(1 + dist_params_len)];
120 }
121 array[primary_params_len] real primary_params;
122 if (primary_params_len) {
123 int theta_len = size(theta);
124 primary_params = theta[(theta_len - dist_params_len + 1):theta_len];
125 }
126
127 // Compute adjusted delay
128 real d_adj = d - x;
129
130 // Compute log probabilities
131 real log_cdf = dist_lcdf(d_adj | params, dist_id);
132 real log_primary_pdf = primary_dist_lpdf(
133 x | primary_dist_id, primary_params, 0, pwindow
134 );
135
136 if (is_inf(D)) {
137 // No truncation
138 return exp(log_cdf + log_primary_pdf);
139 } else {
140 // Truncate at D
141 real D_adj = D - x;
142 real log_cdf_D = dist_lcdf(D_adj | params, dist_id);
143 return exp(log_cdf - log_cdf_D + log_primary_pdf);
144 }

362vector primary_censored_sone_lpmf_vectorized(
363 int max_delay, data real D, int dist_id,
364 array[] real params, data real pwindow,
365 int primary_dist_id, array[] real primary_params,
366 int approx_truncation
367) {

369 int upper_interval = max_delay + 1;
370 vector[upper_interval] log_pmfs;
371 vector[upper_interval] log_cdfs;
372 real log_normalizer;
373
374 // Check if D is at least max_delay + 1
375 if (D < upper_interval) {
376 reject("D must be at least max_delay + 1");
377 }
378
379 // Compute log CDFs
380 if (approx_truncation) {
381 for (d in 1:upper_interval) {
382 log_cdfs[d] = primary_censored_dist_lcdf(
383 d | dist_id, params, pwindow, positive_infinity(), primary_dist_id,
384 primary_params
385 );
386 }
387 } else {
388 for (d in 1:upper_interval) {
389 log_cdfs[d] = primary_censored_dist_lcdf(
390 d | dist_id, params, pwindow, D, primary_dist_id, primary_params
391 );
392 }
393 }
394
395 // Compute log normalizer using upper_interval
396 if (approx_truncation) {
397 if (D > upper_interval) {
398 if (is_inf(D)) {
399 log_normalizer = 0; // No normalization needed for infinite D
400 } else {
401 log_normalizer = primary_censored_dist_lcdf(
402 upper_interval | dist_id, params, pwindow, positive_infinity(),
403 primary_dist_id, primary_params
404 );
405 }
406 } else {
407 log_normalizer = log_cdfs[upper_interval];
408 }
409 } else {
410 log_normalizer = 0; // No external normalization for exact truncation
411 }
412
413 // Compute log PMFs
414 log_pmfs[1] = log_cdfs[1] - log_normalizer;
415 for (d in 2:upper_interval) {
416 log_pmfs[d] = log_diff_exp(log_cdfs[d], log_cdfs[d-1]) - log_normalizer;
417 }
