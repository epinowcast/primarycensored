345 vector[upper_interval] log_pmfs;
346 vector[upper_interval] log_cdfs;

348
349 // Check if D is at least max_delay + 1
350 if (D < upper_interval) {
351 reject("D must be at least max_delay + 1");
352 }
353
354 // Compute log CDFs
355 for (d in 1:upper_interval) {

357 d | dist_id, params, pwindow, positive_infinity(), primary_dist_id,
358 primary_params
359 );
360 }
361
362 // Compute log normalizer using upper_interval
363 if (D > upper_interval) {
364 if (is_inf(D)) {
365 log_normalizer = 0; // No normalization needed for infinite D
366 } else {
367 log_normalizer = primary_censored_dist_lcdf(
368 D | dist_id, params, pwindow, positive_infinity(),
369 primary_dist_id, primary_params
370 );
371 }
372 } else {
373 log_normalizer = log_cdfs[upper_interval];
374 }
375
376 // Compute log PMFs
377 log_pmfs[1] = log_cdfs[1] - log_normalizer;
378 for (d in 2:upper_interval) {
