113 // Extract distribution parameters

122 }

137 // Truncate at D

140 return exp(log_cdf - log_cdf_D + log_primary_pdf);

354vector primary_censored_sone_lpmf_vectorized(
355 int max_delay, data real D, int dist_id,
356 array[] real params, data real pwindow,
357 int primary_dist_id, array[] real primary_params,
358 int approx_truncation
359) {
360
361 int upper_interval = max_delay + 1;
362 vector[upper_interval] log_pmfs;
363 vector[upper_interval] log_cdfs;
364 real log_normalizer;
365
366 // Check if D is at least max_delay + 1
367 if (D < upper_interval) {
368 reject("D must be at least max_delay + 1");
369 }
370
371 // Compute log CDFs
372 if (approx_truncation) {
373 for (d in 1:upper_interval) {
374 log_cdfs[d] = primary_censored_dist_lcdf(
375 d | dist_id, params, pwindow, positive_infinity(), primary_dist_id,
376 primary_params
377 );
378 }
379 } else {
380 for (d in 1:upper_interval) {
381 log_cdfs[d] = primary_censored_dist_lcdf(
382 d | dist_id, params, pwindow, D, primary_dist_id, primary_params
383 );
384 }
385 }
386
387 // Compute log normalizer using upper_interval
388 if (approx_truncation) {
389 if (D > upper_interval) {
390 if (is_inf(D)) {
391 log_normalizer = 0; // No normalization needed for infinite D
392 } else {
393 log_normalizer = primary_censored_dist_lcdf(
394 upper_interval | dist_id, params, pwindow, positive_infinity(),
395 primary_dist_id, primary_params
396 );
397 }
398 } else {
399 log_normalizer = log_cdfs[upper_interval];
400 }
401 } else {
402 log_normalizer = 0; // No external normalization for exact truncation
403 }
404
405 // Compute log PMFs
406 log_pmfs[1] = log_cdfs[1] - log_normalizer;
407 for (d in 2:upper_interval) {
408 log_pmfs[d] = log_diff_exp(log_cdfs[d], log_cdfs[d-1]) - log_normalizer;
409 }
410
411 return log_pmfs;
412}
